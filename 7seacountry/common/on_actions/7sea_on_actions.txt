on_actions = {
	on_weekly = {
    	effect = {
        	if = {
            	limit = { 
					original_tag = ITA
            	    is_ai = no
					NOT = {
						has_global_flag = ITA_played
					}
            	}
				set_global_flag = ITA_played
        	}
			else = {
				clr_global_flag = ITA_played
			}
			if = {
            	limit = { 
					original_tag = GER
            	    is_ai = no
					NOT = {
						has_global_flag = GER_played
					}
            	}
				set_global_flag = GER_played
        	}
			else = {
				clr_global_flag = GER_played
			}
			if = {
            	limit = { 
					original_tag = ENG
            	    is_ai = no
					NOT = {
						has_global_flag = ENG_played
					}
            	}
				set_global_flag = ENG_played
        	}
			else = {
				clr_global_flag = ENG_played
			}
			if = {
            	limit = { 
					original_tag = FRA
            	    is_ai = no
					NOT = {
						has_global_flag = FRA_played
					}
            	}
				set_global_flag = FRA_played
        	}
			else = {
				clr_global_flag = FRA_played
			}
			if = {
            	limit = { 
					original_tag = SOV
            	    is_ai = no
					NOT = {
						has_global_flag = SOV_played
					}
            	}
				set_global_flag = SOV_played
        	}
			else = {
				clr_global_flag = SOV_played
			}
			if = {
				limit = {
					any_county = {
						is_ai = no
						has_war_with = GER
					}
					NOT = {
						has_global_flag = has_war_with_GER
					}
				}
				set_global_flag = has_war_with_GER
			}
			else_if ={
				limit = {
					has_global_flag = has_war_with_GER
				}
				clr_global_flag = has_war_with_GER
			}

			if = {
				limit = {
					any_county = {
						is_ai = no
						has_war_with = ENG
					}
					NOT = {
						has_global_flag = has_war_with_ENG
					}
				}
				set_global_flag = has_war_with_ENG
			}
			else_if ={
				limit = {
					has_global_flag = has_war_with_ENG
				}
				clr_global_flag = has_war_with_ENG
			}
    	}
	}
}
